// Set the target date and time const targetDate = new Date('December 09, 2024 13:55:00').getTime(); // Update the countdown every second const countdown = setInterval(() => { const now = new Date().getTime(); const timeLeft = targetDate - now; // Calculate hours, minutes, and seconds const totalHours = Math.floor(timeLeft / (1000 * 60 * 60)); // Total hours left const minutes = Math.floor((timeLeft % (1000 * 60 * 60)) / (1000 * 60)); const seconds = Math.floor((timeLeft % (1000 * 60)) / 1000); // Format numbers to always show two digits const formatNumber = (num) => (num < 10 ? '0' + num : num); // Update the HTML document.getElementById('days').style.display = 'none'; // Hide the days section document.getElementById('hours-value').innerText = formatNumber(totalHours); document.getElementById('minutes-value').innerText = formatNumber(minutes); document.getElementById('seconds-value').innerText = formatNumber(seconds); // If the countdown is over, stop the timer if (timeLeft < 0) { clearInterval(countdown); document.querySelector('.timer-title').innerText = "The countdown is over!"; document.querySelector('.timer').style.display = 'none'; } }, 1000);
